# Sort from File
add numbers.txt -txt
upload numbers.txt numbers.txt
sort.ScriptSortDataFromTxt numbers.txt sorted.txt -create -clear
	sort.parse_file_by_buckets  numbers.txt common -create -clear
	sort.reduce_identity_common_key common sorted -clear -create
	sort.export_sequence_common_key sorted sorted.txt  -clear -create
download sorted.txt sorted.txt

# Sort from Rand
sort.ScriptSortDataFromGenerator sorted.txt -create -clear
	set test.number_of_samples		100000000
	sort.random_sequence_generator numbers -create -clear
	sort.reduce_to_common_key numbers common -clear -create
	sort.reduce_identity_common_key common sorted -clear -create
	sort.export_sequence_common_key sorted sorted.txt  -clear -create
download sorted.txt sorted.txt

# And to create the input data file
set test.number_of_samples		100000000
sort.random_sequence_generator numbers -create -clear
sort.export_sequence numbers numbers.txt  -clear -create
download numbers.txt numbers.txt
